final class GoldNoiseRenderer {

    init?(width: Int = 150, height: Int = 150)

    /// Renders one frame of gold noise asynchronously.
    /// Completion handler called on a Metal callback thread with the IOSurface

### ClockService API (modified by S4P-3, no consumer changes needed):
```swift
@MainActor
final class ClockService: ObservableObject {
    init()          // sets initial state, does NOT start timer
    func resume()   // refCount++, starts timer if refCount == 1
    func pause()    // refCount--, stops timer if refCount == 0
}
```
