## Interface Contracts

### games.json schema addition (P0 output, P1 consumes)
```json
{
  "white": "...", "black": "...", "whiteElo": "...", "blackElo": "...",
  "tournament": "...", "year": 1986,
  "mateBy": "white",
  "finalMove": "e7e8q",
  "moveSequence": ["e7e8q", "d5d6", "c4c5", "b3b4", "a2a3", "h7h6",
                   "g6g7", "f5f6", "e4e5", "d3d4", "c2c3", "b1b2"],
  "positions": ["fen_1_before_mate", ..., "fen_12_before_mate"]
}
```
- `moveSequence[i]` = UCI move played FROM `positions[i]`
- `moveSequence[0]` = `finalMove` (the checkmate move from positions[0])
- `moveSequence[i]` reaches `positions[i-1]` for i > 0
- Always exactly 12 entries

### ChessGame model (P1 output, P2+P3+P4 consume)
```swift
struct ChessGame: Codable {
    let white: String
    let black: String
    let whiteElo: String       // "?" if unknown
    let blackElo: String
    let tournament: String
    let year: Int
    let month: String?
    let round: String?
    let mateBy: String         // "white" or "black"
    let finalMove: String      // UCI, e.g. "e7e8q"
    let moveSequence: [String] // NEW: 12 UCIs, moveSequence[i] from positions[i]
    let positions: [String]    // 12 FEN strings

    init(white:black:whiteElo:blackElo:tournament:year:
         month:round:mateBy:finalMove:moveSequence:positions:)
    // moveSequence defaults to [] and round/month/mateBy/finalMove have defaults
}
```

### PuzzleEngine public API (P2 output, P3+P4 consume)
```swift
struct PuzzleEngine {
    enum SubmitResult {
        case correctContinue(opponentMoves: [(uci: String, fen: String)])
        case success
        case wrong(triesRemaining: Int, resetAutoPlays: [(uci: String, fen: String)])
        case failed
    }

    let game: ChessGame
    let hour: Int
    private(set) var currentPositionIndex: Int   // indexes into game.positions
    private(set) var triesUsed: Int              // 1..3
    private(set) var isComplete: Bool
    private(set) var succeeded: Bool

    var startPositionIndex: Int { hour - 1 }
    var currentFEN: String { game.positions[currentPositionIndex] }
    var isUserTurn: Bool  // true when mating-side's turn at currentFEN
    var expectedMove: String { game.moveSequence[currentPositionIndex] }

    init(game: ChessGame, hour: Int)
    mutating func advancePastOpponentMoves() -> [(uci: String, fen: String)]
    mutating func submit(uci: String) -> SubmitResult
}
```

### GuessService public API (P3 output, P4 consumes)
```swift
struct PuzzleResult: Codable {
    let succeeded: Bool
    let triesUsed: Int    // 1, 2, or 3
}

struct PuzzleStats: Codable {
    var winsOnFirstTry: Int
    var winsOnSecondTry: Int
    var winsOnThirdTry: Int
    var losses: Int
    var totalPlayed: Int { winsOnFirstTry + winsOnSecondTry + winsOnThirdTry + losses }
}

@MainActor final class GuessService: ObservableObject {
    @Published private(set) var engine: PuzzleEngine?       // nil = not started or done
    @Published private(set) var result: PuzzleResult?       // non-nil = puzzle done this hour
    @Published private(set) var stats: PuzzleStats          // all-time
    @Published private(set) var currentHourKey: String

    var hasResult: Bool { result != nil }
    var secondsUntilNextHour: Int { ... }

    init(clockService: ClockService)
    func startPuzzle(game: ChessGame, hour: Int) -> [(uci: String, fen: String)]?
    func submitMove(uci: String) -> PuzzleEngine.SubmitResult?
}
```
